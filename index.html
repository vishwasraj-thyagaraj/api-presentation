hasMany & belongsTo Relationship

with inverse


							// post.js
							export default class Post extends Model {
							  @attr('string') title;
							  @attr('string') description;
							  @belongsTo('user', { inverse: "posts" }) user;
							}

							// user.js
							export default class User extends Model {
							  @attr('string') name;
							  @hasMany('post', { inverse: "user" }) posts;
							}

Without inverse (Ember does its best to find relationships)


								// post.js
								export default class Post extends Model {
								  @attr('string') title;
								  @attr('string') description;
								  @belongsTo('user') user;
								}

								// user.js
								export default class User extends Model {
								  @attr('string') name;
								  @hasMany('post') posts;
								}

Setting inverse to null removes the relationship


							// post.js
							export default class Post extends Model {
							  @attr('string') title;
							  @attr('string') description;
							  @belongsTo('user', { inverse: null }) user;
							}

							// user.js
							export default class User extends Model {
							  @attr('string') name;
							  @hasMany('post', { inverse: null }) posts;
							}
Fetching data for hasMany/belongsTo relationships
  • Lazy Loaded
  • Eager Loaded (Side Loaded)
  • Lazy Loaded (ember-data-has-many-query addon + components)

							// Lazy Loaded Example
							// API Response
							{
							  user: { 
							    id: 1,
							    name: 'Vishwas',
							    posts: [1, 2]
							  }
							}

							// models/user.js
							export default class User extends Model {
							  @attr('string') name;
							  @hasMany('post', { async: true }) posts;
							  // make api call using user.posts
							}

							// serializers/user.js
							import ApplicationSerializer as APS from './application';

							export default class UserSerializer extends APS {}
						

							// Eager Loaded
							// API Response
							{
							  user: { 
							    id: 1,
							    name: 'Vishwas',
							    posts: [
							      { id: 1, title: 'Test 1', description: 'Description 1' },
							      { id: 2, title: 'Test 2', description: 'Description 2' }
							    ]
							  }
							}

							// models/user.js
							export default class User extends Model {
							  @attr('string') name;
							  @hasMany('post', { async: false }) posts;
							  // no api call required
							}

							// serializers/user.js
							import ApplicationSerializer as APS from './application';
							import { EmbeddedRecordsMixin as ERM } from '@ember-data/serializer/rest';

							export default class UserSerializer extends APS.extend(ERM) {
							  attrs = {
							    posts : {
							      deserialize: 'records'
							    }
							  }
							}
